# Git Commands to Upload to GitHub

# 1. Configure Git (first time only)
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# 2. Initialize Git repository
git init

# 3. Add all files
git add .

# 4. Create first commit
git commit -m "Initial commit - Weather Dashboard with settings and dark mode"

# 5. Add remote repository (replace YOUR_USERNAME with your GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/weather-dashboard.git

# 6. Push to GitHub
git branch -M main
git push -u origin main

# If you get authentication error, you might need to:
# - Use GitHub Personal Access Token instead of password
# - Or use GitHub Desktop for easier authentication
